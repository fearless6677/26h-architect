// 霍尔三维结构 —— 系统工程的"魔方"框架
// 提出者: A.D.Hall (1969)

const 霍尔三维结构 = {

  时间维: [                // ── "什么时候做"
    "规划", "拟订方案", "研制", "生产",
    "安装", "运行", "更新"
  ],

  逻辑维: [                // ── "怎么想"
    "明确问题", "确定目标", "系统综合",
    "系统分析", "优化", "决策", "实施"
  ],

  知识维: [                // ── "用什么知识"
    "工程", "医学", "建筑", "商业",
    "法律", "管理", "社会科学", "艺术"
  ]

}

// 💡 核心思想：三个维度交叉 = 立体管理

核心：霍尔三维结构由 时间维（7个阶段）、逻辑维（7个步骤）、知识维（多学科） 组成三维空间结构。每个交叉点代表一个具体工作单元，需要调用对应知识维的专业知识。 三个维度共同构成系统工程的全生命周期管理框架。
📐 A.D.Hall 1969年提出 🕐 时间维 7 阶段 🧠 逻辑维 7 步骤 📚 知识维 多学科 🚀 航天/国防/大型工程项目

📝 考试真题

  • 霍尔三维结构以时间维、逻辑维、知识维组成立体结构
  • 在研制阶段会做出研制方案及生产计划
  • 某航天工程项目用霍尔三维结构进行全生命周期管理

✅ 优点

  • 提供了系统化的方法论框架
  • 覆盖工程全生命周期
  • 多学科综合集成
  • 结构清晰，便于管理和分工

❌ 局限性

  • 偏重过程管理，灵活性不足
  • 对复杂迭代关系描述不够
  • 知识维边界在实际中难以明确划分
  • 早期方法论，未涵盖敏捷/迭代思想

📝 考试关键词

霍尔三维结构 时间维 逻辑维 知识维 7个时间阶段 7个逻辑步骤 1969年
